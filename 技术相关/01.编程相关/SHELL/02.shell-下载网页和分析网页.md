#### 下载指定网页的所有img选项
```
for info in $(curl -k -s http://www.eztang.cn/shejituandui/120.html \
             |grep img |grep uploads \
             |awk -F"src=" '{print $2}' |awk -F "\"|'" '{print $2}' 
    );
do
    filename=$(echo -n ${info} |awk -F/ '{print $NF}')
    curl -k -s http://www.eztang.cn${info} > $filename
done
```
