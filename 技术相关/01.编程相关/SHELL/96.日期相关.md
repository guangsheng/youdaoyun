```shell
# 从1月1号大院到12月31日
for i in {0..365}; do
    date_str=$(date -d "2017-01-01 +${i} day" +%Y-%m-%d)
    echo $date_str
done

# 从00:00:00打印到24:00:00
for i in {0..24}; do
    if [ $i -lt 10 ]; then
        time_str="0${i}:00:00"
    else
        time_str="${i}:00:00"
    fi
    echo $time_str
done

```
