#coding=utf-8

def sanitize(timestring):
	s_spliter=''
	if '-' in timestring:
		s_spliter = '-'
	elif ':' in timestring:
		s_spliter = ':'
	else:
		return timestring
	(smin, ssec) = timestring.split(s_spliter)
	return (smin+'.'+ssec)

def print_file(filename):
	try:
		with open(filename) as vfile:
			data = vfile.readline()
		filelist = data.strip().split(',')
	except IOError as err:
  		print('File error: '+str(err))

        #列表推导语法
	new_filelist = [sanitize(each_t) for each_t in filelist]
        #集合：无序无重复项，用set可以将列表转换为集合，进而达到去重的效果
        #[0:3] 分片，只取前三项
	print(sorted(set(new_filelist))[0:3])

print_file('hfpy_ch5_data/james.txt')
print_file('hfpy_ch5_data/julie.txt')
